\noindent \textbf{Q. 1} Say whether the following statements are TRUE or FALSE. If True, prove, otherwise give a counter example:

\begin{enumerate}[label=(\alph*)]
    \item The map $T(x) = \sin(x)$ is a linear map. %\hfill \textbf{[3 marks]} I muted the makrs added to each Question
    
    \item All bounded linear maps are continuous at 0. %\hfill \textbf{[3 marks]}
    
    \item $X$ be a real normed linear space such that $X \neq 0$ then $X^* \neq 0$. %\hfill \textbf{[3 marks]}
    
    \item $l_2([0, 1])$ is a subset of $l_1([0, 1])$. %\hfill \textbf{[3 marks]}
    
    \item A norm is a sublinear function. %\hfill \textbf{[3 marks]}
\end{enumerate}

\vspace{0.3cm}

\noindent \textbf{Q. 2} 
\begin{enumerate}[label=(\alph*)]
    \item Arrange the following spaces in an increasing order of inclusion and justify your answer: $C_0, C, l_2, l_1, l_\infty$.% \hfill \textbf{[7 marks]}
    
    \item Let $X$ be a real linear space and $\|\cdot\|$ be a norm on $X$. Let $\alpha > 0$, show that $\alpha\|\cdot\|$ is also a norm on $X$. %\hfill \textbf{[8 marks]}
\end{enumerate}

\vspace{0.3cm}

\noindent \textbf{Q. 3} 
\begin{enumerate}[label=(\alph*)]
    \item Let the map $T : \mathbb{R}^2 \to \mathbb{R}$ be defined by $T(x, y) = x+y$. Show that $T$ is a bounded linear map.% \hfill \textbf{[8 marks]}
    
    \item Compute the norm of the map $T$ in 3(a) above with respect to norms 1 and 2 in $\mathbb{R}^2$. %\hfill \textbf{[7 marks]}
\end{enumerate}

\noindent \textbf{Q. 4} 
\begin{enumerate}[label=(\alph*)]
    \item State (without proof) the \emph{analytic form} of the Hahn Banach theorem and one of its consequences.% \hfill \textbf{[6 marks]}
    
    \item Let $X$ be a real normed linear space and $x \in X$ such that $f(x) = 0$ $\forall f \in X^*$. Show that $x = 0$, that is, the vector, $x$, is necessarily zero. %\hfill \textbf{[9 marks]}
\end{enumerate}

\noindent \textbf{Q. 5} 
\begin{enumerate}[label=(\alph*)]
    \item Let $x = \left\{1, \frac{1}{2}, \frac{1}{3}, \ldots \right\}$ and $y = \left\{\frac{1}{3}, \frac{1}{4}, \frac{1}{5}, \ldots \right\}$. Verify that $x$ and $y$ are in $l_2$ and compute $\langle x, y\rangle$. %\hfill \textbf{[6 marks]}
    
    \item State and prove the Cauchy Schwartz inequality. %\hfill \textbf{[9 marks]}
\end{enumerate}

\noindent \textbf{Q. 6} Let
\[
C_0 = \left\{x = (x_n)_{n\geq1} : \lim_{n\to\infty}x_n = 0\right\}.
\]

\begin{enumerate}[label=(\alph*)]
    \item Show that $C_0$ is a vector space. %\hfill \textbf{[7 marks]}
    
    \item Show that the function $x \to \|x\|_\infty = \sup_{n\geq0}|x_n|$ is a norm on $C_0$. %\hfill \textbf{[8 marks]}
\end{enumerate}
